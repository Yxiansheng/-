继承

1. 原型链继承
```
function Father () {
    this.name = "father"
}

function Child (age) {
    this.age = age
}
Child.prototype = new Father()
```       
缺点：无法传递参数给父类

****

  2. 实例继承
 function Father () {
            this.name = "father"
        }

        function Child (age) {
            let f = new Father()
            f.age = age
            return f
        }
  3. 构造函数继承  
  function Father () {
            this.name = "father"
        }

        function Child (age) {
            Father.apply(this, arguments)
            this.age = age
        }
4. 拷贝继承
 function Father () {
            this.name = "father"
        }

        function Child (age) {
            let f = new Father()

            for (let key in f) {
                this[key] = f[key]
            }
        }
  5.  组合继承
 function Father () {
            this.name = "father"
        }

        function Child (age) {
            Father.apply(this, arguments)
            this.age = age
        }
        Child.prototype = new Father()
        Child.prototype.constructor = Child

  6.  寄生组合继承
 function Father () {
            this.name = "father"
        }

        function Child (age) {
            Father.apply(this, arguments)
            this.age = age
        }
        (function () {
            let f = function () {
            }
            f.prototype = Father.prototype
            Child.prototype = new f()
            Child.prototype.constructor = Child
        })()
