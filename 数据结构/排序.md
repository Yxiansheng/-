#### 冒泡排序

const bubbleSort = a => {
  if (a.length <= 1) return a
  const arr = a.slice()
  const len = arr.length

  for (let i = 0; i < len - 1; i++) {
    let flag = true // 每一轮设置该轮有无比较标志
    for (let j = 0; j < len - i; j++) {
      if (arr[j] > arr[j + 1]) {
        const temp = arr[j]
        arr[j] = arr[j + 1]
        arr[j + 1] = temp
        flag = false 
      } 
      if (flag) { break } // 该轮结束无变化，说明已排序完毕，无需再次进行比较
    }
  }

  return arr
}
