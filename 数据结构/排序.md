#### 冒泡排序
```
const bubbleSort = a => {
  if (a.length <= 1) return a
  const arr = a.slice()
  const len = arr.length

  for (let i = 1; i < len; i++) {
    let flag = true // 每一轮设置该轮有无比较标志
    for (let j = 0; j < len - i; j++) {
      if (arr[j] > arr[j + 1]) {
        [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]]
        flag = false 
      } 
    }
    if (flag) break // 该轮结束无变化，说明已排序完毕，无需再次进行比较
  }

  return arr
}
```

#### 选择排序
```
const selectSort = a => {
  if (a.length <= 1) return a
  const arr = a.slice()
  const len = arr.length

  for (let i = 0; i < len; i++) {
    let min = i
    for (let j = i + 1; j < len; j++) {
      if (arr[min] > arr[j]) {
        min = j
      }
    }
    if (min !== j) {
      [arr[i], arr[min]] = [arr[min], arr[i]]
    }
  }
  return arr
}
```

#### 插入排序
```
const insertSort = a => {
  const arr = a.slice()
  const len = arr.length

  for (let i = 1; i < len; i++) {
    let j = i;
    const temp = arr[i]
    while (j > 0 && temp < arr[j - 1]) {
      arr[j] = arr[j - 1]
      j--
    }
    arr[j] = temp
  }
  return arr
}
```

#### 希尔排序
```
 const shellSort = a => {
  const arr = a.slice()
  const len = arr.length

  let gap = 1;
  while (gap < arr.length) {
    gap = gap * 3 + 1;
  }

  while (gap > 0) {
    for (let i = gap; i < len; i++) {
      let j = i - gap
      const temp = arr[i]

      while (j >= 0 && arr[j] > temp) {
        arr[j + gap] = arr[j]
        j -= gap
      }
      arr[j + gap] = temp
    }
    gap = Math.floor(gap / 3)
  }
  return arr
}
```

#### 归并排序
```
const mergeSort = a => {
  const arr = a.slice()
  const len = arr.length
  if (len < 2) return arr
  const middle = Math.floor(len / 2)
  const left = a.slice(0, middle)
  const right = a.slice(middle)

  return merge(mergeSort(left), mergeSort(right))
}
const merge = (left, right) => {
  const arr = []
  // 左右数组都有数据
  while (left.length && right.length) {
    if (right[0] < left[0]) {
      arr.push(right.shift())
    } else {
      arr.push(left.shift())
    }
  }
  // 左数组还有数据
  while (left.length) {
    arr.push(left.shift())
  }
  // 右数组还有数据
  while (right.length) {
    arr.push(right.shift())
  }
  return arr
}
```
